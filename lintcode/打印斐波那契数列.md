class Fib:
    def __init__(self):
        self.lst = [1,1]
    
    def __len__(self):
        return len(self.lst)
    
    def __call__(self,x):
        if x < len(self.lst):
            return self.lst
    
        for i in range(2,x):
            self.lst.append(self.lst[i-2]+self.lst[i-1])
        return self.lst
    
    def __getitem__(self,idx):
        if idx < 0:
            return None
        
        if idx < len(self):
            return (self.lst[idx])
        
        self(idx+1)
        return(self.lst[idx])
